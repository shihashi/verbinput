\def\@has@carry#1@{%
  \ifnum\@has@one#11@\@has@two#12@\@has@three#13@>0 0\else1\fi}%
  \def\inc#1{\ifnum\@has@carry#1@>0 1\fi\@inc#1@}

  \ifnum\@has@carry#2@>0
    \ifnum#1>3 1\@inc#2@\else \the\numexpr#1+1\relax\@inc#2@\fi
  \else #1\@inc#2@\fi}
